Control de Flujo

if...else

function comprobarNumero(num) {
    if (num > 0) {
        return 'Positivo';
    } else if (num < 0) {
        return 'Negativo';
    } else {
        return 'Cero';
    }
}
console.log(comprobarNumero(5)); // Positivo

switch

function diaDeLaSemana(dia) {
    switch (dia) {
        case 0: return 'Domingo';
        case 1: return 'Lunes';
        case 2: return 'Martes';
        case 3: return 'Miércoles';
        case 4: return 'Jueves';
        case 5: return 'Viernes';
        case 6: return 'Sábado';
        default: return 'Día inválido';
    }
}
console.log(diaDeLaSemana(2)); // Martes

try/catch/throw

function dividir(a, b) {
    try {
        if (b === 0) throw new Error('División por cero');
        return a / b;
    } catch (error) {
        return error.message;
    }
}
console.log(dividir(10, 0)); // División por cero

Bucles e Iteración

for

for (let i = 0; i < 5; i++) {
    console.log(i); // 0, 1, 2, 3, 4
}

while

let j = 0;
while (j < 5) {
    console.log(j);
    j++;
}

do...while

let k = 0;
do {
    console.log(k);
    k++;
} while (k < 5);

Funciones

Definir y Llamar Funciones

function saludar(nombre) {
    return 'Hola, ' + nombre;
}
console.log(saludar('Oscar')); // Hola, Oscar

Expresiones y Operadores

Asignación y Comparación

let a = 5;
let b = 10;
console.log(a == b); // false
console.log(a != b); // true

Operadores Aritméticos

let suma = a + b; // 15
let resta = b - a; // 5

Operadores Lógicos

console.log(true && false); // false
console.log(true || false); // true

Números y Fechas

Objeto Number

console.log(Number.isInteger(10)); // true

Objeto Math

console.log(Math.sqrt(16)); // 4
console.log(Math.random()); // Número aleatorio

Objeto Date

let fecha = new Date();
console.log(fecha); // Fecha y hora actual

Formateo de Texto

Cadenas Literales

console.log(`Hola, ${nombre}!`); // Hola, Ana!

Objeto String
